CREATE TABLE CROP{
    CROP_ID Number NOT NULL,
    CROP_TYPE VARCHAR(200) NOT NULL,
    CONSTRAINT PK_CROP PRIMARY KEY (CROP_ID)
}


CREATE TABLE GEOGRAFY{
    GEOGRAFY_ID Number NOT NULL,
    COUNTRY VARCHAR(200) NOT NULL,
    FEDERALSTATE VARCHAR(200) NOT NULL,
    CONSTRAINT PK_GEO PRIMARY KEY (GEOGRAFY_ID)
}

CREATE TABLE PRODUCTION_YEAR{
    TIME_ID Number NOT NULL,
    PRODUCTION_YEAR Number NOT NULL,
    CONSTRAINT PK_PROD_Y PRIMARY KEY (TIME_ID)
}

CREATE TABLE CROPPRODUKTION{
    CROP_PRODUCTION_ID Number NOT NULL,
    TIME_ID Number NOT NULL,
    GEOGRAFY_ID Number NOT NULL,
    CROP_ID Number NOT NULL,
    AMMOUNT Number,
    CONSTRAINT FK_CROP_PROD_GEO FOREIGN KEY (GEOGRAFY_ID) REFERENCES GEOGRAFY(GEOGRAFY_ID),
    CONSTRAINT FK_CROP_PROD_Y FOREIGN KEY (TIME_ID) REFERENCES PRODUCTION_YEAR(TIME_ID),
    CONSTRAINT FK_CROP_PROD_CROP FOREIGN KEY (CROP_ID) REFERENCES CROP(CROP_ID),
    CONSTRAINT PK_CROP_PROD PRIMARY KEY (CROP_PRODUCTION_ID)
}
